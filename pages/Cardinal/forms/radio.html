Psk-Radio component is designed to be used inside a psk-radio-group component because in the essence, only one radio can be selected inside a group.

If you want to be able to select multiple options from a list, you can use Psk-Checkboxes or Psk-Select

Radios are typically rendered as small circles, which are filled or highlighted when selected.

Psk-radio-group is notified when a psk-radio is clicked, so the group can then update the model.

 import ContainerController from "../../cardinal/controllers/base-controllers/ContainerController.js"; const model = { gender: { label: "Select your gender", required: true, options: [{ label: "Male", value: 'M' }, { label: "Female", value: "F" } ], value: '' } } export default class FormComponentExampleController extends ContainerController { constructor(element) { super(element); this.model = this.setModel(JSON.parse(JSON.stringify(model))); this.feedbackEmitter = null; this.on('openFeedback', (e) => { e.preventDefault(); e.stopImmediatePropagation(); this.feedbackEmitter = e.detail; }); let radioSubmit = () => { let gender = this.model.gender.value; if (gender === "M"){ this.feedbackEmitter(`Hello there, gentleman!`,"radio Example","alert-primary") } else { this.feedbackEmitter(`Good day to you, mademoiselle!`,"radio Example","alert-primary") } } this.on("Radio submit",radioSubmit,true); } }
